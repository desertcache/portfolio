Agent Orchestration Flow
# Swarm-style agent handoffs with dynamic routing
async def _route_to_next_agent(self, state: AgentState) -> AgentState:
    match state.current_agent:
        case "planner":
            return await self._run_searcher(state)
        case "analyzer":
            if state.needs_more_info:
                return await self._run_searcher(state)  # Loop back
            return await self._run_verifier(state)
        case "verifier":
            if not state.verified:
                return await self._run_analyzer(state)  # Re-analyze
            return await self._run_writer(state)
Hybrid Search with Fusion
# Reciprocal Rank Fusion combining dense + sparse retrieval
def hybrid_search(query: str, k: int = 10) -> List[Document]:
    dense_results = chroma_db.similarity_search(query, k=k*2)
    sparse_results = bm25_index.search(query, k=k*2)
    
    # RRF fusion with cross-encoder reranking
    fused = reciprocal_rank_fusion(dense_results, sparse_results)
    return cross_encoder.rerank(query, fused)[:k]

const generateNotes = (): string => {

  notes += `PROCESS:\n`;

GLSL Shader - Organic Morph
// Smooth 4-second state transitions with organic distortion
vec3 morphPosition(vec3 pos, float t, int state) {
  float noise = snoise(pos * 2.0 + time * 0.3);
  float amplitude = mix(
    stateAmplitudes[prevState],
    stateAmplitudes[state],
    smoothstep(0.0, 1.0, t)  // 4s ease
  );
  return pos + normal * noise * amplitude;
}
